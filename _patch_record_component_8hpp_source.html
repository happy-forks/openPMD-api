69 PatchRecordComponent::load(size_t idx, std::unique_ptr< T >& data)
70 {
71  Datatype dtype = determineDatatype< T >();
72  if( dtype != getDatatype() )
73  throw std::runtime_error("Type conversion during particle patch loading not yet implemented");
74 
75  Extent dse = getExtent();
76  if( dse[0] < idx )
77  throw std::runtime_error("Index does not reside inside patch (patch: " + std::to_string(dse[0])
78  + " - index: " + std::to_string(idx) + ")");
79 
80  if( !data )
81  data = std::unique_ptr< T >(new T);
82  T* raw_ptr = data.get();
83 
84  Parameter< Operation::READ_DATASET > dRead;
85  dRead.offset = {idx};
86  dRead.extent = {1};
87  dRead.dtype = getDatatype();
88  dRead.data = raw_ptr;
89  IOHandler->enqueue(IOTask(this, dRead));
90  IOHandler->flush();
91 }
