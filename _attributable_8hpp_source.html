206 Attributable::setAttribute(std::string const& key, T&& value)
207 {
208  dirty = true;
209  auto it = m_attributes->lower_bound(key);
210  if( it != m_attributes->end() && !m_attributes->key_comp()(key, it->first) )
211  {
212  // key already exists in map, just replace the value
213  it->second = Attribute(value);
214  return true;
215  } else
216  {
217  // emplace a new map element for an unknown key
218  m_attributes->emplace_hint(it,
219  std::make_pair(key, Attribute(value)));
220  return false;
221  }
222 }

236 template< typename T >
237 inline T
238 Attributable::readFloatingpoint(std::string const& key) const
239 {
240  static_assert(std::is_floating_point< T >::value, "Type of attribute must be floating point");
241 
242  T t{0};
243  Attribute a = getAttribute(key);
244  Datatype target_dtype = determineDatatype< T >();
245  if( a.dtype == target_dtype )
246  t = a.get< T >();
247  else
248  {
249  warnWrongDtype(key, a.dtype, target_dtype);
250  using DT = Datatype;
251  switch( a.dtype )
252  {
253  case DT::FLOAT:
254  t = static_cast< T >(a.get< float >());
255  break;
256  case DT::DOUBLE:
257  t = static_cast< T >(a.get< double >());
258  break;
259  case DT::LONG_DOUBLE:
260  t = static_cast< T >(a.get< long double >());
261  break;
262  default:
263  throw std::runtime_error("Unknown floating point datatype.");
264  }
265  }
266  return t;
267 }

297  {
298  case DT::VEC_FLOAT:
299  for( auto const& val : a.get< std::vector< float > >() )
300  vt.push_back(static_cast< T >(val));
301  break;
302  case DT::VEC_DOUBLE:

309  break;
310  default:
